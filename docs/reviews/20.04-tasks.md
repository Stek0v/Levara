```typescript
// src/hooks/useAuthGuard.ts
'use client'
export function useAuthGuard() {
  const router = useRouter()
  const pathname = usePathname()
  useEffect(() => {
    if (pathname === '/login') return
    levara.me().catch((err) => {
      if (err?.status === 401) {
        router.push(`/login?next=${encodeURIComponent(pathname)}`)
      }
    })
  }, [pathname])
}

// src/middleware.ts (Next.js 15)
import { NextResponse, type NextRequest } from 'next/server'
export function middleware(req: NextRequest) {
  const { pathname } = req.nextUrl
  if (pathname.startsWith('/api') || pathname === '/login') return NextResponse.next()
  const token = req.cookies.get('auth_token')?.value
  if (!token) return NextResponse.redirect(new URL(`/login?next=${pathname}`, req.url))
  return NextResponse.next()
}
export const config = { matcher: ['/((?!_next|favicon|login).*)'] }

// src/lib/api.ts — update handleResponse()
if (res.status === 401 && typeof window !== 'undefined' && !location.pathname.includes('/login')) {
  location.href = `/login?next=${encodeURIComponent(location.pathname)}`
}
```

- Middleware работает на edge — `auth_token` cookie должен быть `HttpOnly` + `Secure` в prod.

```go
// internal/http/ratelimit.go
func RegisterRateLimiters(app *fiber.App, cfg APIConfig) {
    authLimiter := limiter.New(limiter.Config{
        Max: 10, Expiration: time.Minute,
        KeyGenerator: func(c *fiber.Ctx) string { return c.IP() },
        LimitReached: func(c *fiber.Ctx) error {
            metrics.RateLimitRejected.WithLabelValues("http", "auth").Inc()
            return c.Status(429).JSON(fiber.Map{"error": "rate limit exceeded"})
        },
    })
    app.Post("/auth/login", authLimiter)
    app.Post("/auth/register", authLimiter)

    userLimiter := limiter.New(limiter.Config{
        Max: 100, Expiration: time.Minute,
        KeyGenerator: func(c *fiber.Ctx) string {
            if uid, _ := c.Locals("user_id").(string); uid != "" { return "user:" + uid }
            return "ip:" + c.IP()
        },
    })
    app.Use(userLimiter) // after auth middleware
}

// internal/grpc/ratelimit_interceptor.go
type peerLimiters struct {
    mu sync.Mutex
    m  map[string]*rate.Limiter
    rps rate.Limit
    burst int
}
func (p *peerLimiters) get(addr string) *rate.Limiter { /* LRU */ }

func UnaryRateLimitInterceptor(pl *peerLimiters) grpc.UnaryServerInterceptor {
    return func(ctx context.Context, req any, info *grpc.UnaryServerInfo,
        handler grpc.UnaryHandler) (any, error) {
        pr, _ := peer.FromContext(ctx)
        if !pl.get(pr.Addr.String()).Allow() {
            metrics.RateLimitRejected.WithLabelValues("grpc", "peer").Inc()
            return nil, status.Error(codes.ResourceExhausted, "rate limit exceeded")
        }
        return handler(ctx, req)
    }
}
```

- MCP session-based limit легко обойти созданием новой сессии — дополнительно per-user лимит всё равно сработает.

## T16. WAL recovery order (data-correctness bug)

**Priority:** P0 | **Effort:** M (1–2 дня) | **Risk:** High (data loss)

### Current state
- `internal/store/db.go:122-147` — 2-pass recovery:
  1. Прогоняет WAL, собирая `deletedIDs` (все op=Delete).
  2. При op=Insert проверяет `if _, deleted := deletedIDs[id]; deleted { return }` — пропускает запись.
- **Баг:** сценарий `Insert(id=1, v1) → Delete(id=1) → Insert(id=1, v2)`:
  - Первый pass: `deletedIDs = {1}`.
  - Второй pass: Insert(v1) — пропущен (1 в deletedIDs), Delete — обработан, Insert(v2) — **тоже пропущен** (1 в deletedIDs).
  - Итог: id=1 **исчезает**, хотя должен быть v2.
- `internal/store/wal.go:18-22` — WAL entries не содержат sequence number.

### Target state
- Single-pass sequential replay: WAL читается в порядке записи, каждая операция применяется немедленно.
  - Insert — всегда добавляет (перезатирает существующий).
  - Delete — удаляет если есть.
- `deletedIDs` map полностью убирается.
- WAL format не меняется — только recovery logic.

### Implementation steps
1. Прочитать текущий `db.go:122-147` полностью (сейчас у меня есть высокоуровневое описание).
2. Удалить `deletedIDs := make(map[string]struct{})`.
3. Изменить switch:
   ```go
   switch op {
   case OpInsert:
       // Always apply — WAL order is authoritative
       newLoc, err := db.disk.Write(meta)
       if err != nil { log; return }
       db.insertInMemory(id, vector, newLoc)
   case OpDelete:
       if idx, ok := db.index[id]; ok {
           delete(db.index, id)
           if int(idx) < len(db.revIndex) { db.revIndex[idx] = "" }
           delete(db.metaLocs, idx)
       }
   }
   ```

```go
// internal/store/db.go (обновлённый recovery)
func (db *Levara) recoverFromWAL() error {
    insertCount, deleteCount := 0, 0
    return wal.RecoverEx(func(op byte, id string, vector []float32, meta []byte, loc FileLocation) {
        switch op {
        case OpInsert:
            newLoc, writeErr := db.disk.Write(meta)
            if writeErr != nil {
                fmt.Fprintf(os.Stderr, "WAL recovery: disk write failed for %s: %v\n", id, writeErr)
                return
            }
            db.insertInMemory(id, vector, newLoc)
            insertCount++
        case OpDelete:
            if idx, ok := db.index[id]; ok {
                delete(db.index, id)
                if int(idx) < len(db.revIndex) {
                    db.revIndex[idx] = ""
                }
                delete(db.metaLocs, idx)
                deleteCount++
            }
        }
    })
}
```

```tsx
// src/hooks/use-levara.ts
export function useUpdateSettings() {
    const qc = useQueryClient()
    return useMutation({
        mutationFn: (data: Partial<Settings>) => levara.updateSettings(data),
        onMutate: async (data) => {
```

```go
// internal/grpc/interceptors.go
type ctxUserIDKey struct{}

func UnaryAuthInterceptor(secret string) grpc.UnaryServerInterceptor {
    return func(ctx context.Context, req any, info *grpc.UnaryServerInfo, handler grpc.UnaryHandler) (any, error) {
        if publicMethods[info.FullMethod] { return handler(ctx, req) }
        md, ok := metadata.FromIncomingContext(ctx)
        if !ok { return nil, status.Error(codes.Unauthenticated, "missing metadata") }
        auth := md.Get("authorization")
        if len(auth) == 0 { return nil, status.Error(codes.Unauthenticated, "no auth header") }

        p, ok := auth.VerifyJWT(token, secret)
        if !ok { return nil, status.Error(codes.Unauthenticated, "invalid token") }
        ctx = context.WithValue(ctx, ctxUserIDKey{}, p.Sub)
        return handler(ctx, req)
    }
}
```

## T17. Prom metrics для оставшихся endpoints

**Priority:** P2 | **Effort:** S-M (1–2 дня) | **Risk:** Low

### Current state
- `internal/metrics/telemetry.go` — есть Search/Insert/MCP/Embed/LLM/Sync/Router metrics.
- **Не инструментированы:** Datasets CRUD, Memories CRUD, Users, Auth, Settings, Notebooks, Ontologies, Prune — 30+ endpoints.

### Target state
- Единый `PromInstrumentationMiddleware(operation string)` Fiber middleware.
- Labels: `operation`, `status` (success/error), `method`, **`user_id`** (решение D14).
- **⚠ Cardinality mitigation** (D14): `user_id` НЕ берётся raw из JWT. Применяется bounded-label rule:
  - **Top-50 active users** (по request count за последний час) — реальный `user_id`.
  - Остальные → `user_id="other"`.
  - Обновление top-50 списка — раз в минуту в background goroutine.
  - Alternative: Prometheus recording rules агрегируют per-tier (free/pro/enterprise).
- Histograms: `<endpoint>_duration_seconds`.
- Counters: `<endpoint>_requests_total{status, user_id}`.

### Implementation steps
1. Создать `internal/metrics/user_bucket.go` — bounded user-label tracker:
   ```go
   type UserBucket struct {
       mu       sync.RWMutex
       topN     int // 50
       counts   map[string]int64      // userID → request count (rolling)
       topSet   map[string]struct{}   // current top-N userIDs
   }
   func (b *UserBucket) Label(userID string) string {
       if userID == "" { return "anon" }
       b.mu.RLock(); defer b.mu.RUnlock()
       if _, ok := b.topSet[userID]; ok { return userID }
       return "other"
   }
   func (b *UserBucket) Observe(userID string) { /* increment count */ }
   // Background goroutine: каждую минуту пересчитывает topSet по counts, resets
   ```
2. Расширить `internal/metrics/telemetry.go`:
   ```go
   var (
       HTTPRequests = promauto.NewCounterVec(
           prometheus.CounterOpts{Name: "levara_http_requests_total"},
           []string{"operation", "status", "user_id"},
       )
       HTTPDuration = promauto.NewHistogramVec(
           prometheus.HistogramOpts{Name: "levara_http_duration_seconds"},
           []string{"operation", "user_id"},
       )
       UserBucketInstance = NewUserBucket(50)
   )
   ```
3. Создать `internal/http/middleware.go`:
   ```go
   func PromInstrumentationMiddleware(operation string) fiber.Handler {
       return func(c *fiber.Ctx) error {
           start := time.Now()
           err := c.Next()
           status := "success"
           if err != nil || c.Response().StatusCode() >= 400 { status = "error" }
           uid, _ := c.Locals("user_id").(string)
           metrics.UserBucketInstance.Observe(uid)
           label := metrics.UserBucketInstance.Label(uid)
           metrics.HTTPRequests.WithLabelValues(operation, status, label).Inc()
           metrics.HTTPDuration.WithLabelValues(operation, label).Observe(time.Since(start).Seconds())
           return err
       }
   }
   ```
4. Применить middleware к route groups:
   ```go
   datasets := app.Group("/datasets", PromInstrumentationMiddleware("datasets"))
   // ...
   ```
5. Добавить gauge `levara_user_bucket_size` для мониторинга cardinality.
6. Grafana dashboard: добавить panels для top-50 users + общий summary (+`user_id="other"` breakdown).

### DoD
- [ ] `HTTPRequests` + `HTTPDuration` metrics добавлены с labels `operation`, `status`, `user_id`.
- [ ] `UserBucket` реализован: top-50 whitelist + `"other"` fallback + background refresh.
- [ ] Middleware применён ко всем route groups.
- [ ] `/metrics` endpoint содержит metrics для всех endpoint групп.
- [ ] Cardinality bounded: max 51 значений user_id на endpoint (top-50 + "other" + "anon").
